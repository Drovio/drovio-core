Window Frame

A window frame is a simple desktop-like window that can present information to the user. It is used with popups in order to show a window popup.

// Create and show a simple window frame to the user
	
// Use the class
use \UI\Presentation\frames\windowFrame;

// Create a frame
$frame = new windowFrame($id = "");

// This is a UIObject, so we build
$frame->build($title = "Window Frame", $class = "windowFrame");

// We can append anything we like in the frame
$header = DOM::create("h2", "This is a header in the frame");
$frame->append($header);

$p = DOM::create("p", "This is a paragraph in the frame");
$frame->append($p);

// After we finish building, we return the frame report
// This will return a persistent popup
return $frame->getFrame();
