Toggler

A toggler serves as a flexible ui object what includes information hidden from the user to save space. It has a header which if clicked shows or hides the hidden information.

// Create a toggler
	
// Use the class
use \UI\Presentation\togglers\toggler;

// Create a toggler object
$toggler = new toggler();

// This is a UIObject, so we build
// It supports the chain pattern, so we get the element immediately
$title = "User Information";
$togBody = DOM::create("div", "", "", "userInfoContainer");
$togItem = $toggler->build($id = "tog_id", $title, $togBody, $open = FALSE)->get();

// Append to body
DOM::append($body, $togItem);


// We can append anything in the toggler body
$header = DOM::create("h2", "User Information in details");
DOM::append($togBody, $header);

$description = DOM::create("p", "User description in details", "", "user_desc");
DOM::append($togBody, $description);
