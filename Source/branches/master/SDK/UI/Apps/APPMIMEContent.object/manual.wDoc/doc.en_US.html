Application MIME Async Content Builder

This class extends the MIMEContent interface to enable connection to application files. It allows to download application files in account or in team mode.

// Building a custom MIME content
	
// Use the class
use \UI\Apps\APPMIMEContent;

// Create the object
$mimeContent = new APPMIMEContent();


// Set the application relative file to be downloaded
$mimeContent->set("/reports/daily_report.txt", $type = APPMIMEContent::CONTENT_APP_STREAM, $mode = APPMIMEContent::TEAM_MODE);


// Get the report immediately to download the file
return $mimeContent->getReport($suggestedFileName = "", $ignore_user_abort = FALSE, $removeFile = FALSE);
